Add `customLovelaceModules` to the existing `services.home-assistant` block in `modules/automation/home-assistant.nix`:

```nix
# Frigate Lovelace card (frigate-hass-card) for camera dashboard
# Package renamed from frigate-hass-card to advanced-camera-card in v7.0.0
customLovelaceModules = with pkgs.home-assistant-custom-lovelace-modules; [
  advanced-camera-card
];
```

Place this after the existing `customComponents` declaration.

Next, add a Lovelace YAML dashboard configuration. Keep the default dashboard in storage mode (UI-editable) and add a new "cameras" dashboard in YAML mode via `services.home-assistant.config.lovelace`:

```nix
lovelace = {
  mode = "storage";  # Keep default dashboard UI-editable
  dashboards = {
    cameras = {
      mode = "yaml";
      filename = "cameras-dashboard.yaml";
      title = "Cameras";
      icon = "mdi:cctv";
      show_in_sidebar = true;
      require_admin = false;
    };
  };
};
```

Do NOT add an explicit `lovelace.resources` block. The NixOS HA module auto-registers JS resources from `customLovelaceModules` at `/local/nixos-lovelace-modules/`. Adding explicit resources would create duplicates.

Add a tmpfiles rule to the existing `systemd.tmpfiles.rules` list to create the empty dashboard file so HA does not crash on boot:

```nix
"f /var/lib/hass/cameras-dashboard.yaml 0644 hass hass"
```

Run `make check` to verify the configuration builds without errors.

<task type="auto">
  <name>Task 2: Write cameras dashboard YAML with event filtering and detection badges, then deploy</name>
  <files>modules/automation/home-assistant.nix</files>
  <action>
Write the cameras dashboard content as a Nix expression that generates JSON (valid YAML) and deploy it to `/var/lib/hass/cameras-dashboard.yaml` on ser8.

**Approach (definitive):** Use `builtins.toJSON` on a Nix attrset to produce a JSON file (JSON is valid YAML). Deploy via `systemd.tmpfiles.rules` using `L+` to symlink the Nix store path into `/var/lib/hass/`. HA YAML-mode dashboards read the file at startup and on reload -- symlinks work fine for read-only access.

Create a `let` binding at the top of the module (inside the existing function args `{ config, lib, pkgs, ... }:` block):

```nix
let
  camerasDashboard = pkgs.writeText "cameras-dashboard.yaml" (builtins.toJSON dashboardConfig);

  # Helper: frigate-hass-card (advanced-camera-card) for live camera view
  # Per user decision: live video stream default, click to fullscreen, no bounding boxes
  # Includes conditional "Detection Off" badge overlay when detection switch is off
  cameraCard = cameraName: displayName: {
    type = "custom:frigate-card";
    cameras = [
      {
        camera_entity = "camera.${cameraName}";
        live_provider = "auto";
        title = displayName;
        frigate = {
          camera_name = cameraName;
        };
      }
    ];
    live = {
      controls = {
        builtin = true;
      };
    };
    menu = {
      buttons = {
        frigate = { enabled = true; };
        fullscreen = { enabled = true; };
      };
    };
    view = {
      default = "live";
    };
    # "Detection Off" overlay badge when detection is disabled (per user decision)
    elements = [
      {
        type = "conditional";
        conditions = [
          {
            entity = "switch.${cameraName}_detect";
            state = "off";
          }
        ];
        elements = [
          {
            type = "state-badge";
            entity = "switch.${cameraName}_detect";
            style = {
              top = "8%";
              right = "8%";
              left = "auto";
              color = "red";
              opacity = "0.8";
            };
          }
        ];
      }
    ];
  };

  # Helper: frigate-hass-card in event/gallery mode for event browsing
  # Per user decision: thumbnail grid, 7-day history, inline expand
  # Shows events filtered by specific object types for a single camera
  eventGalleryCard = cameraName: displayName: {
    type = "custom:frigate-card";
    cameras = [
      {
        camera_entity = "camera.${cameraName}";
        title = displayName;
        frigate = {
          camera_name = cameraName;
        };
      }
    ];
    view = {
      default = "clips";
    };
    event_gallery = {
      min_columns = 3;
    };
  };

  dashboardConfig = {
    views = [
      # Tab 1: Live Cameras
      {
        title = "Live Cameras";
        path = "live";
        icon = "mdi:camera";
        cards = [
          # 3-camera grid (equal layout per user decision, fixed order: driveway, front_door, garage)
          {
            type = "grid";
            columns = 3;
            square = false;
            cards = [
              (cameraCard "driveway" "Driveway")
              (cameraCard "front_door" "Front Door")
              (cameraCard "garage" "Garage")
            ];
          }
          # Detection Controls section below camera grid (per user decision: separate section)
          {
            type = "entities";
            title = "Detection Controls";
            show_header_toggle = true;
            entities = [
              { entity = "switch.driveway_detect"; name = "Driveway Detection"; }
              { entity = "switch.front_door_detect"; name = "Front Door Detection"; }
              { entity = "switch.garage_detect"; name = "Garage Detection"; }
              { entity = "switch.driveway_motion"; name = "Driveway Motion"; }
              { entity = "switch.front_door_motion"; name = "Front Door Motion"; }
              { entity = "switch.garage_motion"; name = "Garage Motion"; }
            ];
          }
        ];
      }
      # Tab 2: Events -- filterable by camera AND object type (per user decision)
      # Uses frigate-hass-card's built-in gallery view which provides:
      # - Thumbnail grid of detection snapshots with camera name + timestamp
      # - Clicking thumbnail expands inline with larger snapshot and event details
      # - Card-level media filter for object types
      # Separate cards per camera for camera-level filtering; each card's built-in
      # filter bar allows filtering by object type (person, car, package)
      {
        title = "Events";
        path = "events";
        icon = "mdi:motion-sensor";
        cards = [
          # All cameras combined view with object type filtering
          {
            type = "custom:frigate-card";
            cameras = [
              {
                camera_entity = "camera.driveway";
                title = "Driveway";
                frigate.camera_name = "driveway";
              }
              {
                camera_entity = "camera.front_door";
                title = "Front Door";
                frigate.camera_name = "front_door";
              }
              {
                camera_entity = "camera.garage";
                title = "Garage";
                frigate.camera_name = "garage";
              }
            ];
            view = {
              default = "clips";
            };
            event_gallery = {
              min_columns = 3;
            };
            # The frigate-hass-card gallery view includes a built-in filter bar
            # at the top allowing filtering by camera and by object type (label).
            # With all 3 cameras listed, the filter bar shows camera + label dropdowns.
            media_gallery = {
              controls = {
                filter = {
                  mode = "left";
                };
              };
            };
          }
        ];
      }
    ];
  };
in
```
